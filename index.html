 cin>>rx>>ry;
 //Region 1
 p=ry*ry-rx*rx*ry+rx*rx/4;
 x=0;y=ry;
 while(2.0*ry*ry*x <= 2.0*rx*rx*y)
 {
if(p < 0)
{
x++;
p = p+2*ry*ry*x+ry*ry;
}
else
{
x++;y--;
p = p+2*ry*ry*x-2*rx*rx*y-ry*ry;
}
putpixel(xc+x,yc+y,RED);
putpixel(xc+x,yc-y,RED);
putpixel(xc-x,yc+y,RED);
putpixel(xc-x,yc-y,RED);
 }

 //Region 2

 while(y > 0)
 {
if(p <= 0)
{
x++;y--;
p = p+2*ry*ry*x-2*rx*rx*y+rx*rx;
}
else OUTPUT
{
y--;
p = p-2*rx*rx*y+rx*rx;
}

 }
